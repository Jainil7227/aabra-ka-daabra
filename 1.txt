#include <stdio.h>
#include <stdlib.h>

// Function for Linear Search
void ls() 
{
    int n, key, flag = 0;
    printf("Enter the total number of students present in training session : ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter the roll numbers of students present: ");
    for (int i = 0; i < n; i++) 
    {
        scanf("%d", &arr[i]);
    }
    printf("Enter roll number to be searched: ");
    scanf("%d", &key);

    for (int i = 0; i < n; i++) 
    {
        if (arr[i] == key) 
        {
            printf("Roll number %d is present in class.\n", key);
            flag = 1;
            break;
        }
    }

    if (flag == 0) 
    {
        printf("Roll number %d is absent.\n", key);
    }
}

// Function for Binary Search
void bs() 
{
    int n, key, flag = 0;
    printf("Enter the total number of students present in training session in order: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter the roll numbers of students present: ");
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter roll number to be searched: ");
    scanf("%d", &key);

    int low = 0, high = n - 1, mid;
    while (low <= high) {
        mid = (low + high) / 2;

        if (arr[mid] == key) {
            printf("Roll number %d is present in the class.\n", key);
            flag = 1;
            break;
        } else if (arr[mid] < key) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    if (flag == 0) {
        printf("Roll number %d is absent.\n", key);
    }
}

int main() {
    int s;
    printf("Attendance Monitoring System\nMain Menu:\n1. Linear Search\n2. Binary Search\nEnter your selection: ");
    scanf("%d", &s);
    switch(s)
    {
    case 1: ls(); break;
    case 2: bs(); break;
    }
    return 0;
}
