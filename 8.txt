#include <stdio.h>
#include <stdlib.h>

struct node 
{
    int data;
    struct node *left, *right;
};

struct node* newnode(int value)
{
    struct node* temp = (struct node*)malloc(sizeof(struct node));
    temp->data = value;
    temp->left = temp->right = NULL;
    return temp;
}

struct node* insert(struct node* root, int value)
{
    if (root == NULL)
    {
        return newnode(value);
    }
    else if (value > root->data)
    {
        root->right = insert(root->right, value);
    }
    else 
    {
        root->left = insert(root->left, value);
    }
    return root;
}

int hieght(struct node* root)
{
    if (root == NULL)
    {
        return 0;
    }
    int left = hieght(root->left);
    int right = hieght(root->right);
    return (left > right ? left : right) + 1;
}

int findmin(struct node* root)
{
    if (root == NULL)
    {
        printf("Tree is empty!\n");
        return -1;
    }
    struct node* current = root;
    while (current->left != NULL)
    {
        current = current->left;
    }
    return current->data;
}

void mirror(struct node* root)
{
    if (root == NULL)  // ✅ Added base case
    {
        return;
    }
    struct node* temp = root->left;
    root->left = root->right;
    root->right = temp;
    mirror(root->left);
    mirror(root->right);
}

int search(struct node* root, int key)
{
    if (root == NULL)  // ✅ Added NULL check
    {
        return 0;
    }
    if (root->data == key)
    {
        return 1;
    }
    if (key > root->data)  // ✅ Fixed comparison
    {
        return search(root->right, key);  // ✅ Added return
    }
    else 
    {
        return search(root->left, key);  // ✅ Added return
    }
}

void inorder(struct node* root)
{
    if (root != NULL)
    {
        inorder(root->left);
        printf("%d ", root->data);
        inorder(root->right);
    }
}

int main()
{
    int choice, value;
    struct node* root = NULL;
    int values[] = {50, 30, 70, 20, 40, 60, 80};
    int n = sizeof(values) / sizeof(values[0]);
    
    for (int i = 0; i < n; i++)
    {
        root = insert(root, values[i]);  // ✅ Fixed: update root
    }
    
    printf("Initial Inorder traversal: ");
    inorder(root);
    printf("\n\nMain Menu:\n");
    printf("1. Insert\n2. Find longest path length\n3. Find minimum value\n");
    printf("4. Mirror the tree\n5. Search for a value\n6. Exit\n");
    
    choice = 0;  // ✅ Initialize choice
    while (choice != 6)
    {
        printf("\nEnter your choice: ");
        scanf("%d", &choice);
        
        switch (choice)
        {
            case 1:
                printf("Enter value to insert: ");
                scanf("%d", &value);
                root = insert(root, value);  // ✅ Fixed: update root
                printf("Inorder traversal after insertion: ");
                inorder(root);
                printf("\n");
                break;
                
            case 2:
                printf("Longest path length: %d\n", hieght(root));
                break;
                
            case 3:
                {
                    int min = findmin(root);
                    if (min != -1)
                        printf("Minimum value in the tree: %d\n", min);
                }
                break;
                
            case 4:
                mirror(root);
                printf("Inorder traversal after mirroring: ");
                inorder(root);
                printf("\n");
                break;
                
            case 5:
                printf("Enter value to search: ");
                scanf("%d", &value);
                if (search(root, value))
                    printf("Value %d found in the tree.\n", value);
                else
                    printf("Value %d not found in the tree.\n", value);
                break;
                
            case 6:
                printf("Exiting program.\n");
                break;
                
            default:
                printf("Invalid choice. Please try again.\n");
        }
    }
    
    return 0;
}











Algorithm
Start with root = NULL.

1.For each value in the input sequence, call root = insert(root, value) to build the BST.

2.To insert a value: if root == NULL create a new node; else if value < root->data recurse left; else recurse right.

3.To display (optional), perform an inorder traversal to visit nodes in sorted order.

5.To insert a new node at runtime, call the same insert(root, value) and update root if needed.

6.To compute number of nodes in the longest path (height in nodes), return 1 + max(height(left), height(right)) with base case 0 for NULL.

7.To find the minimum value, follow current = current->left until current->left == NULL and return current->data.

8.To mirror the tree, at each node swap left and right then recursively mirror both children.

9.To search for a value, compare with root->data: if equal return true; if smaller recurse left; else recurse right; return false at NULL.

10.Always handle NULL (empty tree) cases in every function and return appropriate default/indicator values (e.g., not found or tree empty).
