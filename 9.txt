#include <stdio.h>
int n;

void dfs(int v,int visited[],int adj[n][n])
{
    visited[v]=1;
    for (int i=0;i<n;i++)
    {
        if (adj[v][i] !=0 && !visited[i])
        {
            dfs(i,visited,adj);
        }
    }
}

int isconnected(int adj[n][n])
{
    int visited[n];
    for (int i=0;i<n;i++)
    {
        visited[i]=0;
    }
    dfs(0,visited,adj);
    for (int i=0;i<n;i++)
    {
    if (!visited[i])
        {
            return 0;
        }
    }
    return 1;
}

int main()
{
    int edges;
    printf("Enter number of cities : ");
    scanf("%d",&n);
    int adj[n][n];
    for (int i=0;i<n;i++)
    {
        for (int j=0;j<n;j++)
        {
            adj[i][j]=0;
        }
    }
    printf("Enter the number of flight connections : ");
    scanf("%d",&edges);
    printf("Enter source destination cost :\n");
    for (int i=0;i<edges;i++)
    {
        int src,dest,cost;
        scanf("%d %d %d",&src,&dest,&cost);
        adj[src][dest]=cost;
    }

    if (isconnected(adj))
    {
        printf("The flight network(graph) is connected . ");
    }
    else {printf("The flight network(graph) is not connected . ");}
    return 0;
}

🔟 In dfs(v), mark city v as visited.
1️⃣1️⃣ For every city i, if adj[v][i] != 0 and visited[i] == 0, call dfs(i).
1️⃣2️⃣ After DFS completes, check all entries of visited[].
1️⃣3️⃣ If all cities are visited → graph is connected.
