//Write a function for sorting array of floating point number in ascending order using selection sort, bubble sort and display the top 5 score.
#include <stdio.h>
void bubble(float arr[],int n)
{
    int temp;
    for (int i=0;i<n-1;i++)
    {
        for (int j=0;j<n-i-1;j++)
        {
            if (arr[j]>arr[j+1])
            {
                temp = arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
    printf("The percentage of all students in sorted order are : ");
    for (int k=0;k<n;k++)
    {
        printf("%.2f ",arr[n-k-1]);
    }
    printf("\n");
    printf("The percentage of top 5 students are : ");
    for (int k=0;k<5;k++)
    {
        printf("%.2f ",arr[n-k-1]);
    }
}
void selection(float arr[],int n)
{   
    int min;
    float temp;
    for (int i=0;i<n-1;i++)
    {
        min=i;
        for (int j=i+1;j<n;j++)
        {
            if(arr[j]<arr[min])
            {
                min=j;
            }
        }
        if(min !=i)
        {
            temp=arr[min];
            arr[min]=arr[i];
            arr[i]=temp;
        }
    }
    printf("The percentage of all students in sorted order are : ");
    for (int k=0;k<n;k++)
    {
        printf("%.2f ",arr[n-k-1]);
    }
    printf("\n");
    printf("The percentage of top 5 students are : ");
    for (int k=0;k<5;k++)
    {
        printf("%.2f ",arr[n-k-1]);
    }
}
int main()
{
    int n;
    printf("Enter the total number of students : ");
    scanf("%d",&n);
    float perc[n];
    printf("Enter the first year percentage of students : ");
    for (int i=0;i<n;i++)
    {
        scanf("%f",&perc[i]);
    }
    printf("Bubble sort : \n");
    bubble(perc,n);
    printf("\nSelection sort : \n");
    selection(perc,n);
    return 0;
}
